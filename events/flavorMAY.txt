namespace = flavor_may
normal_or_historical_nations = yes

# Popol Vuh
country_event = {
	id = flavor_may.1
	title = flavor_may.1.t
	desc = flavor_may.1.d
	picture = BIG_BOOK_eventPicture
	
	fire_only_once = yes

	trigger = {
		OR = {
			tag = MAY
			tag = KIC
			tag = ITZ
			tag = XIU
			tag = COC
		}
		religion_group = christian
		any_owned_province = {
			NOT = {
				religion_group = christian
			}
		}
		NOT = {
			has_country_modifier = may_popol_vuh
		}
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = {
		name = flavor_may.1.a
		add_country_modifier = {
			name = "may_popol_vuh"
			duration = 3650
		}
	}
	option = {
		name = flavor_may.1.b
		add_prestige = 5
		add_papal_influence = 2
	}
}

# League of Mayapan
country_event = {
	id = flavor_may.2
	title = flavor_may.2.t
	desc = flavor_may.2.d
	picture = CIVIL_WAR_eventPicture
	
	trigger = {
		religion = mesoamerican_religion
		NOT = { has_country_flag = may_league_mayapan_flag }
	}
	
	mean_time_to_happen = {
		months = 22
	}
	
	immediate = {
		hidden_effect = {
			set_country_flag = may_league_mayapan_flag
		}
	}
	
	option = {
		name = flavor_may.2.a
		add_prestige = 10
		every_country = {
			limit = {
				religion = mesoamerican_religion
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = stronger_alone }
		}
	}
	option = {
		name = flavor_may.2.b
		every_country = {
			limit = {
				religion = mesoamerican_religion
				NOT = { tag = ROOT }
			}
			add_opinion = { who = ROOT modifier = league_mayapan }
		}	
	}
}

# SPAIN GETS CLAIMS
country_event = {
	id = flavor_may.7
	title = flavor_may.7.t
	desc = flavor_may.7.d
	picture = RELIGION_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = flavor_may.7.a
		add_prestige = 10
		add_papal_influence = 2
		FROM = {
			random_owned_province = {
				limit = {
					OR = {
						has_port = yes
						any_neighbor_province = {
							owned_by = SPA
						}
					}
				}
				add_claim = SPA
			}
		}
	}
}
